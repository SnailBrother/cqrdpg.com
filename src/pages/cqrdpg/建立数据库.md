--网站监控
CREATE TABLE RdpgCode.dbo.WebsiteRecord (
    -- 1. 主键
    id INT IDENTITY(1,1) PRIMARY KEY,
    -- 2. 核心标识字段 (用于计算 UV 和去重)
    visitorid NVARCHAR(64) NOT NULL,       -- 客户端生成的唯一ID (如 UUID/Fingerprint)，用于计算 UV
    sessionid NVARCHAR(64) NOT NULL,       -- 会话ID (同一会话内多次请求ID相同)，用于计算 Sessions 和 跳出率
    ipaddress NVARCHAR(45) NOT NULL,       -- 用户IP (支持IPv6)，用于计算 IP数 和地域分析   
    -- 3. 时间字段 (用于时间范围筛选和时长计算)
    visittime DATETIME2 NOT NULL DEFAULT GETDATE(), -- 访问发生的具体时间
    pageloadtime INT,                     -- 页面加载耗时(毫秒)，可选，用于性能分析   
    -- 4. 页面与来源字段 (用于统计 PV, 来路, 入口, 受访)
    currenturl NVARCHAR(2048) NOT NULL,    -- 当前访问的页面 URL (受访页)
    referrerurl NVARCHAR(2048),            -- 来源 URL (HTTP Referer)，用于统计“来路”
    entryurl NVARCHAR(2048),               -- 本次会话的入口 URL (通常在会话第一条记录时写入，或后端推导) 
    -- 5. 设备与环境信息 (可选，用于未来扩展分析)
    useragent NVARCHAR(1024),              -- 浏览器 UA，用于解析设备类型(OS, Browser)
    countrycode CHAR(2),                   -- 国家代码 (可通过IP库解析后存入)
    cityname NVARCHAR(100),                -- 城市名称  
    -- 6. 行为标记 (用于优化查询性能)
    isbounce BIT DEFAULT 1,                -- 是否跳出 (1:是, 0:否)。后端在会话结束时更新此字段
    stayduration INT                       -- 在该页面的停留时长(秒)。如果是单页应用或最后一步，需特殊计算
    -- 索引建议 (非常重要，否则大数据量下查询极慢)
);

-- 创建索引以加速查询
CREATE INDEX IDXVisitTime ON RdpgCode.dbo.WebsiteRecord (visittime);
CREATE INDEX IDXSessionId ON RdpgCode.dbo.WebsiteRecord (sessionid);
CREATE INDEX IDXVisitorId ON RdpgCode.dbo.WebsiteRecord (visitorid);
CREATE INDEX IDXReferrer ON RdpgCode.dbo.WebsiteRecord (referrerurl);

前端采集 (React 端):
你需要编写一个脚本（或在 useEffect 中），在页面加载时生成一个唯一的 visitor_id (存入 localStorage 以便长期识别) 和一个 session_id ( sessionStorage 或 30分钟无操作刷新)。
获取 document.referrer 作为 referrer_url。
获取 window.location.href 作为 current_url。
调用后端 API 将这些数据写入数据库。
后端处理:
入口页判断: 当收到一条新记录时，检查数据库中该 session_id 是否已存在记录。如果不存在，则当前 URL 即为 entry_url。
跳出率计算: 这是一个聚合查询。通常不需要实时存入 is_bounce 字段，而是在查询报表时动态计算：“查找所有 session_id，如果该 ID 在选定时间段内只有 1 条记录，则计为跳出”。
停留时长: 精确计算较难。简单做法是记录进入时间，用户离开时（beforeunload 事件）发送最后一次请求，后端计算差值。或者仅统计多页浏览的时长。
性能优化:
随着数据量增加（PV 增长很快），WebsiteRecord 表会迅速变大。
务必建立上述提到的 索引。
考虑定期归档旧数据（例如将一年前的数据移入历史表），以保证“今日/昨日”查询的速度。

CREATE TABLE RdpgCode.dbo.Suggestion (
    -- 1. 主键：自增ID
    id INT IDENTITY(1,1) PRIMARY KEY,
    -- 2. 核心内容字段
    content NVARCHAR(1000) NOT NULL,       -- 评论内容  
    -- 3. 用户信息 (匿名模式)
    authorname NVARCHAR(50) DEFAULT N'匿名', -- 昵称  ，默认匿名
    authoremail NVARCHAR(100),            -- 邮箱 (前端有输入框，虽不公开但可用作接收回复通知，可选)
    authortelephone NVARCHAR(255),             -- 电话 (前端有输入框，可选)
    -- 4. 系统自动采集的元数据 (防滥用与展示)
    useragent NVARCHAR(500),              -- 浏览器信息 (对应前端 comment.browser，建议后端自动解析存入)
    ipaddress VARCHAR(45),                -- 用户IP (用于限制频繁提交，不展示给前端)
    -- 5. 回复逻辑关键字段
    parentid INT NULL,                    -- 【关键】父评论ID。如果是主评论则为 NULL，如果是回复则指向被回复的评论ID
    -- 注意：你当前的前端结构是嵌套对象 (comment.reply)，但在数据库中通常用 parentid 更灵活。
    -- 如果必须严格匹配前端的一对一回复结构，也可以保留此字段，但推荐用 parentid 实现无限层级回复。
    -- 6. 互动数据
    likes INT DEFAULT 0,                   -- 点赞数  
    -- 7. 时间戳
    createdat DATETIME DEFAULT GETDATE(), -- 评论时间  
    updatedat DATETIME DEFAULT GETDATE()  -- 更新时间 (用于修改评论，可选)
);

-- 建议添加索引以提高排序和查询效率
